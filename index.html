// Aumentado para 10 segundos function startDriverRadar() { clearInterval(pollingDriverInterval); pollingDriverInterval = setInterval(async () => { if(!currentUser || currentUser.role !== 'driver' || !currentUser.isOnline) return; try { const res = await fetch(`${API_URL}?acao=checar_radar_motorista&email=${encodeURIComponent(currentUser.email)}&t=${Date.now()}`); const data = await res.json(); if(data.sucesso && data.pedido) { const p = data.pedido; // VERIFICAÇÃO ANTI-BUG: Se o status for cancelado, limpa a tela do motorista if(p.status === 'cancelado') { resetRideUI(); return; } currentPedidoId = p.id; currentRideData = p; if(p.status === 'pendente') { if(currentUser.isBusy == 1) { document.getElementById('driver-queue-alert').style.display = 'block'; } else { if(document.getElementById('new-ride-alert').style.display !== 'block') { document.getElementById('req-origin').innerText = p.origem + " (Formiga)"; document.getElementById('req-destination').innerText = p.destino + " (Formiga)"; document.getElementById('req-distance').innerText = p.distancia + ' km'; document.getElementById('req-suggested-price').innerText = 'R$ ' + p.preco.replace('.', ','); document.getElementById('driver-input-price').value = p.preco; document.getElementById('no-rides-msg').style.display = 'none'; document.getElementById('new-ride-alert').style.display = 'block'; } } } else if (['aceito', 'a_caminho', 'chegou', 'em_viagem'].includes(p.status)) { if(document.getElementById('driver-active-ride').style.display !== 'block') { document.getElementById('driver-waiting-panel').style.display = 'none'; document.getElementById('driver-active-ride').style.display = 'block'; document.getElementById('active-pass-avatar').innerHTML = ``; document.getElementById('active-pass-name').innerText = p.passNome; document.getElementById('active-pass-phone-text').innerHTML = ` ${p.passTelefone || 'Sem Telefone'}`; document.getElementById('btn-call-passenger').href = `tel:${p.passTelefone}`; document.getElementById('driver-persistent-dest').innerText = p.destino + ", Formiga - MG"; } rideStatusState = p.status; syncDriverUI(); } } else { if(document.getElementById('new-ride-alert').style.display === 'block') { document.getElementById('new-ride-alert').style.display = 'none'; document.getElementById('no-rides-msg').style.display = 'block'; } document.getElementById('driver-queue-alert').style.display = 'none'; } } catch(e) {} }, 10000); // 10 segundos } // Passageiro Polling também em 10 segundos function startPassengerPolling() { clearInterval(pollingPassInterval); pollingPassInterval = setInterval(async () => { if(!currentPedidoId) return; try { const res = await fetch(`${API_URL}?acao=checar_status_pedido&id=${currentPedidoId}&t=${Date.now()}`); const data = await res.json(); if(data.sucesso && data.pedido) { const p = data.pedido; // Se o passageiro cancelou no banco, o motorista para de responder, pass fecha a tela if(p.status === 'cancelado') { alert("Corrida cancelada."); resetRideUI(); return; } // Se o motorista aceitou, muda a tela do passageiro pra corrida if(p.status === 'aceito' && document.getElementById('modal-analyzing').style.display === 'flex') { document.getElementById('modal-analyzing').style.display = 'none'; document.getElementById('passenger-queue-ride').style.display = 'none'; document.getElementById('passenger-active-ride').style.display = 'block'; document.getElementById('active-driver-avatar').innerHTML = ``; document.getElementById('active-driver-name').innerText = p.driverNome; document.getElementById('active-driver-moto').innerText = p.driverMoto + " | " + p.driverPlaca; document.getElementById('active-driver-rating-display').innerText = parseFloat(p.driverRating || 5).toFixed(1); document.getElementById('pass-active-price').innerText = 'R$ ' + p.preco.replace('.', ','); } // Atualiza status e gps if(['aceito', 'a_caminho', 'chegou', 'em_viagem'].includes(p.status)) { rideStatusState = p.status; syncPassengerUI(); } } } catch(e) {} }, 10000); // 10 segundos }
